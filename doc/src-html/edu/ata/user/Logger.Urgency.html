001package edu.ata.user;
002
003import com.sun.squawk.microedition.io.FileConnection;
004import edu.ata.main.DriverstationInfo;
005import edu.wpi.first.wpilibj.DriverStationLCD;
006import java.io.DataInputStream;
007import java.io.DataOutputStream;
008import java.io.IOException;
009import javax.microedition.io.Connector;
010
011/**
012 * Logging class. Used to give messages to user in different ways depending on
013 * urgency.
014 *
015 * <p> There is a fake {@link Urgency} class that holds urgency types. As we use
016 * java source 1.3/1.4, we cannot use enums.
017 *
018 * <p> Note that all logs are automatically sent to file:///log.txt unless
019 * otherwise set. See {@link Logger#setFileLoggingOn(boolean)}.
020 *
021 * @author Joel Gallant
022 */
023public class Logger {
024
025    private static final String PATH = "file:///log.txt";
026    private static boolean fileLoggingOn = true;
027    private static int lineNum = 1;
028    private static FileConnection logFile;
029
030    /**
031     * Class representing the urgency of the log. The reason for this
032     * implementation is to ensure that logging does not encounter an error when
033     * looking at its urgency. (Ex. int could be out of range, etc.) This keeps
034     * the only possible values to be pre-approved by this class. (No errors
035     * when logging)
036     */
037    public final static class Urgency {
038
039        private Urgency() {
040        }
041        /**
042         * Urgent message. Displayed on DriverStation.
043         */
044        public final static Urgency URGENT = new Urgency();
045        /**
046         * Warning message. Displayed on DriverStation.
047         */
048        public final static Urgency WARNING = new Urgency();
049        /**
050         * User message. Displayed on DriverStation.
051         */
052        public final static Urgency USERMESSAGE = new Urgency();
053        /**
054         * Status report for logs. Is not displayed to user.
055         */
056        public final static Urgency STATUSREPORT = new Urgency();
057        /**
058         * Log message. Is not displayed to user.
059         */
060        public final static Urgency LOG = new Urgency();
061    }
062
063    /**
064     * Logs a message, and depending on the urgency given, reacts appropriately.
065     *
066     * <p> By default, sends messages to the log file at file:///log.txt.
067     *
068     * @param urgency Urgency of the log. Urgencies are available as static
069     * variables in the {@link Logger Log} class.
070     * @param msg Message to send to the user.
071     */
072    public static void log(Urgency urgency, String msg) {
073        String usrMsg = null;
074        if (urgency == Urgency.URGENT) {
075            usrMsg = "Urgent@" + DriverstationInfo.getMatchTime() + "- " + msg;
076            displayLCDMessage(usrMsg);
077        } else if (urgency == Urgency.WARNING) {
078            usrMsg = "Warning@" + DriverstationInfo.getMatchTime() + "- " + msg;
079            displayLCDMessage(usrMsg);
080        } else if (urgency == Urgency.USERMESSAGE) {
081            usrMsg = "User Message - " + msg;
082            displayLCDMessage(usrMsg);
083        } else if (urgency == Urgency.STATUSREPORT) {
084            usrMsg = "Status@" + DriverstationInfo.getMatchTime() + "- " + msg;
085        } else if (urgency == Urgency.LOG) {
086            usrMsg = "Log@" + DriverstationInfo.getMatchTime() + "- " + msg;
087        }
088        System.out.print(usrMsg.concat("\n"));
089        try {
090            if (fileLoggingOn) {
091                logFile(msg.concat("\n"));
092            }
093        } catch (IOException ex) {
094            System.err.println("!!!ERROR WHILE WRITING TO LOG FILE!!!\n" + ex.getMessage());
095        }
096    }
097
098    /**
099     * Logs a message to the log file.
100     *
101     * @param msg message to write
102     * @throws IOException thrown when writing throws an error
103     */
104    public static void logFile(String msg) throws IOException {
105        if (logFile == null) {
106            logFile = (FileConnection) Connector.open(PATH, Connector.READ_WRITE);
107        }
108        appendToFile(msg, logFile);
109    }
110
111    /**
112     * Adds a new part to a text file.
113     *
114     * @param msg message to add
115     * @param fileConnection connection to the file
116     * @throws IOException thrown when problem writing occurs
117     */
118    public static void appendToFile(String msg, FileConnection fileConnection) throws IOException {
119        fileConnection.create();
120        DataOutputStream outputStream = fileConnection.openDataOutputStream();
121        byte[] data = msg.getBytes();
122        try {
123            outputStream.write(data, 0, data.length);
124        } finally {
125            outputStream.close();
126        }
127    }
128
129    /**
130     * Gets the contents of the log file. Goes through each character and
131     * concats it to the string. Is inefficient, so feel free to make it work
132     * faster (<i>Test changes, reading without parsing is difficult</i>).
133     *
134     * @return contents of log
135     * @throws IOException thrown when problem connecting occurs
136     */
137    public static String getLog() throws IOException {
138        if (logFile == null) {
139            logFile = (FileConnection) Connector.open(PATH, Connector.READ_WRITE);
140        }
141        return getTextFromFile(logFile);
142    }
143
144    /**
145     * Returns text from a file, using {@link FileConnection}.
146     *
147     * @param fileConnection connection to the file
148     * @return string representation of contents of the file
149     * @throws IOException thrown when file occurs a problem
150     */
151    public static String getTextFromFile(FileConnection fileConnection) throws IOException {
152        fileConnection.create();
153        DataInputStream inputStream = fileConnection.openDataInputStream();
154        String tmp = "";
155        byte cur;
156        try {
157            while ((cur = (byte) inputStream.read()) > 0) {
158                tmp += (char) cur;
159            }
160            return tmp;
161        } finally {
162            inputStream.close();
163        }
164    }
165
166    /**
167     * Displays a message on the DriverStation LCD. Is convenience method for
168     * {@link DriverStationLCD}.
169     *
170     * <p> This method should usually not be called.
171     * {@link Logger#log(edu.ata.user.Logger.Urgency, java.lang.String)}
172     * provides a better solution to displaying information to the user. This
173     * method is available for public use just in case (It can't hurt).
174     *
175     * @param msg message to display to the user
176     */
177    public static void displayLCDMessage(String msg) {
178        DriverStationLCD.Line line;
179        switch (lineNum) {
180            case (1):
181                line = DriverStationLCD.Line.kUser1;
182                break;
183            case (2):
184                line = DriverStationLCD.Line.kUser2;
185                break;
186            case (3):
187                line = DriverStationLCD.Line.kUser3;
188                break;
189            case (4):
190                line = DriverStationLCD.Line.kUser4;
191                break;
192            case (5):
193                line = DriverStationLCD.Line.kUser5;
194                break;
195            case (6):
196                line = DriverStationLCD.Line.kUser6;
197                break;
198            default:
199                line = DriverStationLCD.Line.kUser1;
200        }
201        DriverStationLCD.getInstance().println(line, lineNum, msg + "\n");
202        DriverStationLCD.getInstance().updateLCD();
203        lineNum++;
204        if (lineNum > 6) {
205            lineNum = 1;
206        }
207    }
208
209    /**
210     * Sets {@link Logger#log(edu.ata.user.Logger.Urgency, java.lang.String)} to
211     * automatically send messages to the log file (file:///log.txt). By
212     * default, is on.
213     *
214     * @param fileLoggingOn whether to log files
215     */
216    public static void setFileLoggingOn(boolean fileLoggingOn) {
217        Logger.fileLoggingOn = fileLoggingOn;
218    }
219
220    /**
221     * Returns whether
222     * {@link Logger#log(edu.ata.user.Logger.Urgency, java.lang.String)} is
223     * automatically sending messages to the log file (file:///log.txt). By
224     * default, is on.
225     *
226     * @return whether logs are being saved to the log file
227     */
228    public static boolean isFileLoggingOn() {
229        return fileLoggingOn;
230    }
231}
