001package edu.ata.main;
002
003import edu.wpi.first.wpilibj.DriverStation;
004
005/**
006 * Contains all DriverStation static methods and other useful methods used for
007 * finding the status of the robot / driverstation.
008 *
009 * <p> <i>Please note:</i> <b>This class is not the same as
010 * {@link DriverStation}. That is the direct link to the actual driverstation.
011 * This is a convenience class with static methods referencing that class.</b>
012 *
013 * @author Joel Gallant
014 */
015public final class DriverstationInfo {
016
017    private static final DriverStation DS = DriverStation.getInstance();
018
019    /**
020     * Gets the location of the team's driver station controls.
021     *
022     * @return the location of the team's driver station controls: 1, 2, or 3
023     */
024    public static int getAllianceLocation() {
025        return DS.getLocation();
026    }
027
028    /**
029     * Return the team number that the Driver Station is configured for
030     *
031     * @return The team number
032     */
033    public static int getTeamNumber() {
034        return DS.getTeamNumber();
035    }
036
037    /**
038     * Return the DS packet number. The packet number is the index of this set
039     * of data returned by the driver station. Each time new data is received,
040     * the packet number (included with the sent data) is returned.
041     *
042     * @return The DS packet number.
043     */
044    public static int getPacketCount() {
045        return DS.getPacketNumber();
046    }
047
048    /**
049     * Is the driver station attached to a Field Management System? Note: This
050     * does not work with the Blue DS.
051     *
052     * @return True if the robot is competing on a field being controlled by a
053     * Field Management System
054     */
055    public static boolean FMSattached() {
056        return DS.isFMSAttached();
057    }
058
059    /**
060     * Read the battery voltage from the specified AnalogChannel.
061     *
062     * This accessor assumes that the battery voltage is being measured through
063     * the voltage divider on an analog breakout.
064     *
065     * @return The battery voltage.
066     */
067    public static double getBatteryVoltage() {
068        return DS.getBatteryVoltage();
069    }
070
071    /**
072     * Return the approximate match time The FMS does not currently send the
073     * official match time to the robots This returns the time since the enable
074     * signal sent from the Driver Station At the beginning of autonomous, the
075     * time is reset to 0.0 seconds At the beginning of teleop, the time is
076     * reset to +15.0 seconds If the robot is disabled, this returns 0.0 seconds
077     *
078     * <p> <b> Warning: This is not an official time (so it cannot be used to argue
079     * with referees) </b>
080     *
081     * @return Match time in seconds since the beginning of autonomous
082     */
083    public static double getMatchTime() {
084        return DS.getMatchTime();
085    }
086
087    /**
088     * Gets a string value of which game mode robot should be in.
089     *
090     * @return String "Teleop" , "Autonomous" or "Disabled".
091     */
092    public static String getGamePeriod() {
093        if (DS.isEnabled()) {
094            if (DS.isOperatorControl()) {
095                return "Teleop";
096            } else if (DS.isAutonomous()) {
097                return "Autonomous";
098            } else {
099                return "Disabled";
100            }
101        } else {
102            return "Disabled";
103        }
104    }
105
106    /**
107     * The Alliance name.
108     *
109     * <p> <i>"Red"</i>, <i>"Blue"</i> or <i>"invalid"</i>.
110     *
111     * @return Alliance name
112     */
113    public static String getAllianceName() {
114        return DS.getAlliance().name;
115    }
116}
