001package edu.ata.automation.dispatch;
002
003/**
004 * Basic class representing game modes. Handles running by itself.
005 *
006 * <p> Implemented by running in a new separate thread. Instantiates a new
007 * thread each time it is started. This is done because threads are implicitly
008 * only runnable once. This creates the need to re-create the {@link Thread}
009 * object each time. Another bonus of this is that priority of the thread is
010 * adjustable between runs if needed. ({@link GameMode#setPriority(int)}).
011 *
012 * @author joel
013 */
014public abstract class GameMode implements Runnable {
015
016    private final String name;
017    private Thread thread;
018    private int priority;
019
020    /**
021     * Creates the game mode with a name. Priority is set to
022     * {@link Thread#NORM_PRIORITY}.
023     *
024     * @param name name of the thread
025     */
026    public GameMode(String name) {
027        this(name, Thread.NORM_PRIORITY);
028    }
029
030    /**
031     * Creates the game mode with a name and priority.
032     *
033     * @param name name of the thread
034     * @param priority priority of the thread (1 - 10)
035     */
036    public GameMode(String name, int priority) {
037        this.name = name;
038        this.priority = priority;
039    }
040
041    /**
042     * Gets the name of the thread.
043     *
044     * @return name of game mode
045     */
046    public final String getName() {
047        return name;
048    }
049
050    /**
051     * Creates a string representation of the thread.
052     *
053     * @return string representation
054     */
055    public final String toString() {
056        return "GameMode:" + name;
057    }
058
059    /**
060     * Checks to see if the thread is still alive. If it is not running / is
061     * null / is not registered as 'alive', returns false.
062     *
063     * @return whether the thread is alive
064     * @see Thread#isAlive()
065     */
066    public final boolean isAlive() {
067        if (thread == null) {
068            return false;
069        }
070        return thread.isAlive();
071    }
072
073    /**
074     * Starts the thread (Game Mode).
075     *
076     * <p> <i>Caution :</i> Thread will not be started if
077     * {@link Thread#isAlive()} does not return false. This prevents the game
078     * mode from running more than once at the same time, but is prone to
079     * concurrency problems.
080     */
081    public final void start() {
082        if (thread != null && thread.isAlive()) {
083            return;
084        }
085        thread = new Thread(this, name);
086        thread.setPriority(priority);
087        thread.start();
088    }
089
090    /**
091     * Sets the priority of the thread. Will take effect the next time you run
092     * the game mode. ({@link GameMode#start()})
093     *
094     * @param priority priority of the thread (1-10)
095     * @see Thread#setPriority(int)
096     */
097    public final void setPriority(int priority) {
098        this.priority = priority;
099    }
100
101    /**
102     * Stops the game mode. Is overridden, and thus cannot be entirely trusted.
103     */
104    public abstract void stop();
105}
