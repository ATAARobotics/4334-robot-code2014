001package edu.ata.automation.dispatch;
002
003/**
004 * {@link GameMode} object designed for autonomous modes.
005 *
006 * @author joel
007 */
008public abstract class AutonomousMode extends InterruptableGameMode implements Autonomous {
009
010    private boolean finished = false;
011
012    /**
013     * Creates the autonomous mode with a name.
014     *
015     * @param name name of the autonomous mode
016     */
017    public AutonomousMode(String name) {
018        super(name, 7);
019    }
020
021    /**
022     * Returns whether the mode is finished running.
023     *
024     * @return whether it is finished.
025     */
026    public final boolean isFinished() {
027        return finished;
028    }
029
030    /**
031     * Method to be run when autonomous mode is run. Handles it's inner methods.
032     */
033    public final void run() {
034        finished = false;
035        init();
036        if (!isInterrupted()) {
037            main();
038        }
039        close();
040        finished = true;
041    }
042}
