001package edu.ata.automation.dispatch;
002
003/**
004 * {@link GameMode} object designed for teleoperated modes.
005 *
006 * @author joel
007 */
008public abstract class TeleoperatedMode extends InterruptableGameMode {
009
010    /**
011     * Method called during teleoperated period in a loop. Is called constantly
012     * during teleoperated period.
013     *
014     * <p> It is recommended that you keep this method short, because
015     * interruptions only occur after this method finishes.
016     */
017    public abstract void loop();
018
019    /**
020     * Creates the mode with a name.
021     *
022     * @param name name of the teleoperated mode
023     */
024    public TeleoperatedMode(String name) {
025        super(name, 8);
026    }
027
028    /**
029     * Method to be run when teleoperated mode is started. Handles it's inner methods.
030     */
031    public void run() {
032        while (!isInterrupted()) {
033            loop();
034        }
035    }
036}
