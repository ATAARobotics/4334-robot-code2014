001package org.scriptreader;
002
003import org.scriptreader.instruction.Method;
004import org.scriptreader.instruction.Returnables;
005import org.scriptreader.values.Condition;
006
007/**
008 * An implicitly casted value, used everywhere that requires a value.
009 *
010 * @author joel
011 */
012public class Value {
013
014    private final Object value;
015    private final String literalValue;
016
017    /**
018     * Creates the value, and implicitly casts it.
019     *
020     * <p>Uses the least specific, but precise enough value type to store the
021     * data.</p> <p>Goes in the order:</p> <p>
022     * <pre>    1.Variables</pre>
023     * <pre>    2.Returnable Method</pre>
024     * <pre>    3.Integer</pre>
025     * <pre>    4.Double</pre>
026     * <pre>    5.Boolean (Literal and Condition)</pre>
027     * <pre>    6.String</pre>
028     *
029     * @param value
030     */
031    public Value(String value) {
032        this.literalValue = value;
033        if (Variables.contains(value)) {
034            this.value = Variables.get(value).getValue();
035        } else if (Returnables.isValid(value) && Returnables.contains(Method.getMethodName(value))) {
036            this.value = Returnables.getFromMethod(value);
037        } else if (isInteger(value)) {
038            this.value = new Integer(Integer.parseInt(value));
039        } else if (isDouble(value)) {
040            this.value = new Double(Double.parseDouble(value));
041        } else if (isBoolean(value)) {
042            this.value = (value.equalsIgnoreCase("true") ? Boolean.TRUE : Boolean.FALSE);
043        } else if (Condition.isCondition(value)) {
044            this.value = (Condition.getConditionFrom(value).isTrue() ? Boolean.TRUE : Boolean.FALSE);
045        } else {
046            if (StringUtils.contains(value, "\"")) {
047                this.value = StringUtils.replace(value, '\"', "");
048            } else {
049                this.value = value;
050            }
051        }
052    }
053
054    /**
055     * Returns the actual value of the {@link Value}.
056     *
057     * @return the actual value
058     */
059    public Object getValue() {
060        return value;
061    }
062
063    /**
064     * Transfers the value to a string.
065     *
066     * @return the value
067     */
068    public String toString() {
069        return getValue().toString();
070    }
071
072    /**
073     * Returns the unique hash code of the value.
074     *
075     * @return the unique hash
076     */
077    public int hashCode() {
078        int hash = 7;
079        hash = 79 * hash + (this.value != null ? this.value.hashCode() : 0);
080        return hash;
081    }
082
083    /**
084     * Determines whether the two values are equal.
085     *
086     * @param obj comparative object
087     * @return whether they are the same value
088     */
089    public boolean equals(Object obj) {
090        if (obj instanceof Value) {
091            return this.value.equals(((Value) obj).value);
092        } else {
093            return false;
094        }
095    }
096
097    /**
098     * Returns the String value that the value started with.
099     *
100     * @return literal value of original value
101     */
102    public String getLiteralValue() {
103        return literalValue;
104    }
105
106    /**
107     * Class method to check if it is possible to transfer a string to an
108     * Integer.
109     *
110     * @param value string to transfer
111     * @return whether string can be transferred
112     */
113    public static boolean isInteger(String value) {
114        try {
115            Integer.parseInt(value);
116            return true;
117        } catch (NumberFormatException ex) {
118            return false;
119        }
120    }
121
122    /**
123     * Class method to check if it is possible to transfer a string to an
124     * Double.
125     *
126     * @param value string to transfer
127     * @return whether string can be transferred
128     */
129    public static boolean isDouble(String value) {
130        try {
131            Double.parseDouble(value);
132            return true;
133        } catch (NumberFormatException ex) {
134            return false;
135        }
136    }
137
138    /**
139     * Class method to check if it is possible to transfer a string to an
140     * Boolean.
141     *
142     * @param value string to transfer
143     * @return whether string can be transferred
144     */
145    public static boolean isBoolean(String value) {
146        return value.equalsIgnoreCase("true") || value.equalsIgnoreCase("false");
147    }
148}
