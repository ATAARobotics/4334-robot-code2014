001package org.scriptreader.blocks;
002
003import org.scriptreader.Instruction;
004import org.scriptreader.InvalidStatementException;
005import org.scriptreader.ScriptReader;
006import org.scriptreader.Statement;
007import org.scriptreader.StringUtils;
008import org.scriptreader.Value;
009
010/**
011 * FOR loop class. Pretty self-explanatory.
012 *
013 * @author joel
014 */
015public class ForLoop extends Instruction {
016
017    /**
018     * Hide this method to use the class. <i><b>This method is meant to be
019     * hidden by it's subclasses. It is called in a tree-like structure down the
020     * children of {@link Statement}.</i></b>
021     *
022     * @param statement the statement to analyze
023     * @return whether it is a valid instruction
024     */
025    public static boolean isValid(String statement) {
026        if (!(StringUtils.contains(statement, "(") && StringUtils.contains(statement, ")"))) {
027            return false;
028        }
029        return statement.startsWith("FOR(") && (statement.endsWith(")") || statement.endsWith("{"));
030    }
031
032    /**
033     * Hide this method to use the class. <i><b>This method is meant to be
034     * hidden by it's subclasses. It is called in a tree-like structure down the
035     * children of {@link Statement}.</i></b>
036     *
037     * <p> It is very important to verify that the statement is valid before
038     * using this method. If it is not valid, it will throw
039     * {@link InvalidStatementException}. </p>
040     *
041     * @param statement the statement to analyze
042     * @return a {@link Statement} object of the type
043     * @throws InvalidStatementException thrown when statement is unrecognizable
044     */
045    public static Statement getStatementFrom(String statement) throws InvalidStatementException {
046        return new ForLoop(new Value(statement.substring(statement.indexOf("(") + 1, statement.indexOf(")"))));
047    }
048    /**
049     * The amount of times to repeat the loop.
050     */
051    public final int repititions;
052
053    /**
054     * Creates a for loop based on one value. Has to be an integer.
055     *
056     * @param value amount of times to loop
057     */
058    public ForLoop(Value value) {
059        this(Integer.parseInt(value.getValue().toString()));
060    }
061
062    /**
063     * Creates a for loop that loops a certain amount of time.
064     *
065     * @param times amount of times to loop
066     */
067    public ForLoop(int times) {
068        super("FOR(" + ")");
069        this.repititions = times;
070    }
071
072    public void run() {
073        this.linesToSkip = linesUntilBreak(fullScript, line);
074        String[] instructions = StringUtils.split(fullScript, ScriptReader.LINE_SEPARATOR);
075        String forLoop = "";
076        int currentLine = line;
077        for (int x = 0; x < linesUntilBreak(fullScript, line); x++) {
078            forLoop += instructions[++currentLine] + ScriptReader.LINE_SEPARATOR;
079        }
080        for (int x = 0; x < repititions; x++) {
081            ScriptReader.runFullScript(forLoop);
082        }
083    }
084}
