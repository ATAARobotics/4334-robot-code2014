001package org.scriptreader.blocks;
002
003import org.scriptreader.Instruction;
004import org.scriptreader.InvalidStatementException;
005import org.scriptreader.ScriptReader;
006import org.scriptreader.Statement;
007import org.scriptreader.StringUtils;
008import org.scriptreader.values.Condition;
009
010/**
011 * WHILE loop. Self-explanatory.
012 *
013 * @author joel
014 */
015public class WhileLoop extends Instruction {
016
017    /**
018     * Hide this method to use the class. <i><b>This method is meant to be
019     * hidden by it's subclasses. It is called in a tree-like structure down the
020     * children of {@link Statement}.</i></b>
021     *
022     * @param statement the statement to analyze
023     * @return whether it is a valid instruction
024     */
025    public static boolean isValid(String statement) {
026        statement = StringUtils.replace(statement, '{', "").trim();
027        return statement.startsWith("WHILE(") && statement.endsWith(")");
028    }
029
030    /**
031     * Hide this method to use the class. <i><b>This method is meant to be
032     * hidden by it's subclasses. It is called in a tree-like structure down the
033     * children of {@link Statement}.</i></b>
034     *
035     * <p> It is very important to verify that the statement is valid before
036     * using this method. If it is not valid, it will throw
037     * {@link InvalidStatementException}. </p>
038     *
039     * @param statement the statement to analyze
040     * @return a {@link Statement} object of the type
041     * @throws InvalidStatementException thrown when statement is unrecognizable
042     */
043    public static Statement getStatementFrom(String statement) throws InvalidStatementException {
044        return new WhileLoop(statement.substring(statement.indexOf("(") + 1, statement.indexOf(")")));
045    }
046    /**
047     * Condition of the while loop. Is checked each loop.
048     */
049    public final String evaluation;
050
051    /**
052     * Creates the while loop with a condition.
053     *
054     * @param evaluation condition of the while loop
055     */
056    public WhileLoop(String evaluation) {
057        super("WHILE(" + evaluation + ")");
058        this.evaluation = evaluation;
059    }
060
061    public void run() {
062        this.linesToSkip = linesUntilBreak(fullScript, line);
063        String[] instructions = StringUtils.split(fullScript, ScriptReader.LINE_SEPARATOR);
064        String whileLoop = "";
065        int currentLine = line;
066        for (int x = 0; x < linesUntilBreak(fullScript, line); x++) {
067            whileLoop += instructions[++currentLine] + System.getProperty("line.separator");
068        }
069        Condition condition = Condition.getConditionFrom(evaluation);
070        while (condition.isTrue()) {
071            ScriptReader.runFullScript(whileLoop);
072        }
073    }
074}
