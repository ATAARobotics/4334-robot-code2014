001package org.scriptreader.blocks;
002
003import org.scriptreader.Instruction;
004import org.scriptreader.InvalidStatementException;
005import org.scriptreader.Statement;
006import org.scriptreader.StringUtils;
007import org.scriptreader.values.Condition;
008
009/**
010 * IF statement. Self-explanatory.
011 *
012 * @author joel
013 */
014public class IfStatement extends Instruction {
015
016    /**
017     * Hide this method to use the class. <i><b>This method is meant to be
018     * hidden by it's subclasses. It is called in a tree-like structure down the
019     * children of {@link Statement}.</i></b>
020     *
021     * @param statement the statement to analyze
022     * @return whether it is a valid instruction
023     */
024    public static boolean isValid(String statement) {
025        statement = StringUtils.replace(statement, '{', "").trim();
026        return statement.startsWith("IF(") && statement.endsWith(")");
027    }
028
029    /**
030     * Hide this method to use the class. <i><b>This method is meant to be
031     * hidden by it's subclasses. It is called in a tree-like structure down the
032     * children of {@link Statement}.</i></b>
033     *
034     * <p> It is very important to verify that the statement is valid before
035     * using this method. If it is not valid, it will throw
036     * {@link InvalidStatementException}. </p>
037     *
038     * @param statement the statement to analyze
039     * @return a {@link Statement} object of the type
040     * @throws InvalidStatementException   thrown when statement is malformed
041     */
042    public static Statement getStatementFrom(String statement) throws InvalidStatementException {
043        return new IfStatement(Condition.getConditionFrom(
044                statement.substring(statement.indexOf("(") + 1, statement.indexOf(")"))));
045    }
046    /**
047     * The condition of the IF statement.
048     */
049    public final Condition condition;
050
051    /**
052     * Creates statement with the condition.
053     *
054     * @param condition condition of the if statement
055     */
056    public IfStatement(Condition condition) {
057        super("IF(" + condition + ")");
058        this.condition = condition;
059    }
060
061    public void run() {
062        if (condition.isTrue()) {
063            this.linesToSkip = 1;
064        } else {
065            this.linesToSkip = linesUntilBreak(fullScript, line);
066        }
067    }
068}
