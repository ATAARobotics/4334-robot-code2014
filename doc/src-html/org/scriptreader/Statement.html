001package org.scriptreader;
002
003/**
004 * Basic statement. Every line of code is a statement of some kind.
005 *
006 * @author joel
007 */
008public class Statement {
009
010    /**
011     * How many lines will be skipped after statement is run.
012     */
013    protected int linesToSkip = 1;
014    /**
015     * Line of code. <p> Set by script reader. Is guaranteed to be the line of
016     * code (By line number / split("\n")). </p>
017     */
018    protected int line;
019    /**
020     * Current script the statement is being run in. <p>Set by script
021     * reader.</p>
022     */
023    protected String fullScript;
024
025    /**
026     * Hide this method to use the class. <i><b>This method is meant to be
027     * hidden by it's subclasses. It is called in a tree-like structure down the
028     * children of {@link Statement}.</i></b>
029     *
030     * @param statement the statement to analyze
031     * @return whether it is a valid instruction
032     */
033    public static boolean isValid(String statement) {
034        return true;
035    }
036
037    /**
038     * Hide this method to use the class. <i><b>This method is meant to be
039     * hidden by it's subclasses. It is called in a tree-like structure down the
040     * children of {@link Statement}.</i></b>
041     *
042     * <p> It is very important to verify that the statement is valid before
043     * using this method. If it is not valid, it will throw
044     * {@link InvalidStatementException}. </p>
045     *
046     * @param statement the statement to analyze
047     * @return a {@link Statement} object of the type
048     * @throws InvalidStatementException thrown when statement is unrecognizable
049     */
050    public static Statement getStatementFrom(String statement) throws InvalidStatementException {
051        // Get rid of errors when tabbing / 4 spacing
052        statement = statement.trim();
053        if (Instruction.isValid(statement)) {
054            return Instruction.getStatementFrom(statement);
055        } else {
056            throw new InvalidStatementException(statement + " is not a statement type.");
057        }
058    }
059
060    /**
061     * Class utility method to determine how many lines exist between the
062     * current line of code and the next break.
063     *
064     * <p>Breaks are defined by '}'</p>
065     *
066     * @param script full script
067     * @param currentLine the line to start from
068     * @return how many lines it is from the current line to the next break
069     * statement ('}')
070     */
071    public static int linesUntilBreak(String script, int currentLine) {
072        String[] s = StringUtils.split(script, ScriptReader.LINE_SEPARATOR);
073        for (int x = currentLine; x < s.length; x++) {
074            if (StringUtils.isEmpty(s[x]) || s[x].startsWith("**")) {
075                // This eleminates errors / comments being processed
076                continue;
077            }
078
079            if (s[x].startsWith("}")) {
080                return x - currentLine;
081            }
082        }
083        return -1;
084    }
085}
