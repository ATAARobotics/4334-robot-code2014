001package org.scriptreader;
002
003import java.util.Hashtable;
004
005/**
006 * Static storage of variables. Stores declared variables.
007 *
008 * @author joel
009 */
010public class Variables {
011
012    private static final Hashtable variables = new Hashtable();
013    
014    /**
015     * Adds a variable to the ram.
016     *
017     * @param name variable name
018     * @param value value of the variable
019     */
020    public static void add(String name, Value value) {
021        variables.put(name, value);
022    }
023
024    /**
025     * Sets a variable in the ram.
026     *
027     * @param name variable name
028     * @param value value of the variable
029     */
030    public static void set(String name, Value value) {
031        variables.put(name, value);
032    }
033
034    /**
035     * Returns the value of a variable in the ram.
036     *
037     * @param name variable name
038     * @return value of the variable
039     */
040    public static Value get(String name) {
041        return (Value) variables.get(name);
042    }
043
044    /**
045     * Returns whether the variable is being stored in the ram.
046     *
047     * @param name variable name
048     * @return if the variable exists in the cache
049     */
050    public static boolean contains(String name) {
051        return variables.containsKey(name);
052    }
053}
