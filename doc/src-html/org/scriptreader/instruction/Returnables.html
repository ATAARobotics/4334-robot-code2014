001package org.scriptreader.instruction;
002
003import java.util.Hashtable;
004import org.scriptreader.Value;
005
006/**
007 * Static class that stores all returnable methods including native methods.
008 *
009 * @author joel
010 */
011public class Returnables {
012
013    private static final Hashtable returnables = new Hashtable();
014
015    static {
016        add(new MathMethod("SUM") {
017            public double doCalc(double num1, double num2) {
018                return num1 + num2;
019            }
020        });
021        add(new MathMethod("SUBTRACT") {
022            public double doCalc(double num1, double num2) {
023                return num1 - num2;
024            }
025        });
026        add(new MathMethod("MULTIPLY") {
027            public double doCalc(double num1, double num2) {
028                return num1 * num2;
029            }
030        });
031        add(new MathMethod("DIVIDE") {
032            public double doCalc(double num1, double num2) {
033                return num1 / num2;
034            }
035        });
036    }
037
038    /**
039     * Adds a returnable object to the cache.
040     *
041     * @param returnable returnable to add
042     */
043    public static void add(Returnable returnable) {
044        returnables.put(returnable.getName(), returnable);
045    }
046
047    /**
048     * Returns whether the cache contains a method
049     *
050     * @param methodName name of the method
051     * @return whether cache contains method
052     */
053    public static boolean contains(String methodName) {
054        return returnables.containsKey(methodName);
055    }
056
057    /**
058     * Returns the returnable method based on it's name.
059     *
060     * @param methodName name of the method
061     * @return returnable
062     */
063    public static Returnable get(String methodName) {
064        return (Returnable) returnables.get(methodName);
065    }
066
067    /**
068     * Returns the value of the returnable based solely on it's method name. Is
069     * not checked. (Use {@link Returnables#isValid(java.lang.String)} to check.
070     *
071     * @param method method name
072     * @return value returned by the method
073     */
074    public static Object getFromMethod(String method) {
075        String[] args = Method.getArguments(method);
076        Value[] trueArgs = new Value[args.length];
077        for (int x = 0; x < args.length; x++) {
078            trueArgs[x] = new Value(args[x].trim());
079        }
080        return get(Method.getMethodName(method)).getValue(trueArgs);
081    }
082
083    /**
084     * Returns whether the method is stored in the cache.
085     *
086     * @param value method name
087     * @return whether the method stored
088     */
089    public static boolean isValid(String value) {
090        return contains(Method.getMethodName(value));
091    }
092
093    /**
094     * Returnable method. Returns a value based on its arguments.
095     */
096    public static interface Returnable {
097
098        /**
099         * Returns the name of the method. Used as Name(Args);
100         *
101         * @return name of the method
102         */
103        public abstract String getName();
104
105        /**
106         * Returns the value based on its arguments.
107         *
108         * @param args arguments given
109         * @return value
110         */
111        public abstract Object getValue(Value[] args);
112    }
113
114    private static abstract class MathMethod implements Returnable {
115
116        public final String name;
117
118        public MathMethod(String name) {
119            this.name = name;
120        }
121
122        public String getName() {
123            return name;
124        }
125
126        public Object getValue(Value[] args) {
127            double value;
128            int start = 0;
129            try {
130                value = Double.parseDouble(args[0].toString());
131                start = 1;
132            } catch (Exception ex) {
133                value = 0;
134            }
135            for (int x = start; x < args.length; x++) {
136                if (Value.isDouble(args[x].toString())) {
137                    value = doCalc(value, Double.parseDouble(args[x].toString()));
138                }
139            }
140            return new Double(value);
141        }
142
143        public abstract double doCalc(double num1, double num2);
144    }
145}
