001package org.scriptreader.instruction;
002
003import org.scriptreader.Instruction;
004import org.scriptreader.InvalidStatementException;
005import org.scriptreader.Statement;
006import org.scriptreader.StringUtils;
007import org.scriptreader.Value;
008import org.scriptreader.Variables;
009import org.scriptreader.blocks.IfStatement;
010import org.scriptreader.blocks.WhileLoop;
011
012/**
013 * The declaration of variables.
014 *
015 * @author joel
016 */
017public class Declaration extends Instruction {
018
019    /**
020     * Hide this method to use the class. <i><b>This method is meant to be
021     * hidden by it's subclasses. It is called in a tree-like structure down the
022     * children of {@link Statement}.</i></b>
023     *
024     * @param statement the statement to analyze
025     * @return whether it is a valid instruction
026     */
027    public static boolean isValid(String statement) {
028        // Ensures first = sign is not == sign
029        return StringUtils.contains(statement, "=") && statement.endsWith(";") && statement.charAt(statement.indexOf("=") + 1) != '='
030                // Cannot be a conditional instruction
031                && !IfStatement.isValid(statement) && !WhileLoop.isValid(statement);
032    }
033
034    /**
035     * Hide this method to use the class. <i><b>This method is meant to be
036     * hidden by it's subclasses. It is called in a tree-like structure down the
037     * children of {@link Statement}.</i></b>
038     *
039     * <p> It is very important to verify that the statement is valid before
040     * using this method. If it is not valid, it will throw
041     * {@link InvalidStatementException}. </p>
042     *
043     * @param statement the statement to analyze
044     * @return a {@link Statement} object of the type
045     * @throws InvalidStatementException thrown when statement is unrecognizable
046     */
047    public static Statement getStatementFrom(String statement) throws InvalidStatementException {
048        return new Declaration(statement.substring(0, statement.indexOf("=")).trim(),
049                new Value(StringUtils.replace(statement.substring(statement.indexOf("=") + 1), ';', "").trim()));
050    }
051    private final String varName;
052    private final Value value;
053
054    /**
055     * Creates the declaration by it's value and name.
056     *
057     * @param varName name of the variable
058     * @param value value of the variable
059     */
060    public Declaration(String varName, Value value) {
061        super(varName + " = " + value);
062        this.varName = varName;
063        this.value = value;
064    }
065
066    public void run() {
067        if (Variables.contains(varName)) {
068            Variables.set(varName, value);
069        } else {
070            Variables.add(varName, value);
071        }
072    }
073}
