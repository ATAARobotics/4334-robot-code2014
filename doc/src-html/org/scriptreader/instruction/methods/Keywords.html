001package org.scriptreader.instruction.methods;
002
003import java.util.Hashtable;
004import org.scriptreader.Value;
005import org.scriptreader.instruction.Method;
006
007/**
008 * Cache that stores all user-defined method. Native methods are stored in
009 * {@link NativeMethods}.
010 *
011 * @author joel
012 */
013public class Keywords {
014
015    private static final Hashtable methods = new Hashtable();
016
017    /**
018     * Adds a user-defined method.
019     *
020     * @param m method to add
021     */
022    public static void add(Method m) {
023        Keywords.methods.put(m.getName(), m);
024    }
025
026    /**
027     * Returns whether a method with the name exists.
028     *
029     * @param name name of the method
030     * @return if it exists in the cache
031     */
032    public static boolean contains(String name) {
033        return methods.containsKey(name);
034    }
035
036    /**
037     * Returns a method based on it's name and arguments. Useful for running
038     * defined methods.
039     *
040     * @param name name of the method
041     * @param args arguments to set the method
042     * @return method based on name
043     * @throws Exception thrown when keyword does not exist
044     */
045    public static Method get(String name, Value[] args) throws Exception {
046        if (!contains(name)) {
047            throw new Exception("The field " + name + " does not exist in Keywords.");
048        }
049        return Method.newMethod((Method) methods.get(name), args);
050    }
051}
