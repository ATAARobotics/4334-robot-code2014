001package org.scriptreader.values;
002
003import org.scriptreader.StringUtils;
004import org.scriptreader.Value;
005
006/**
007 * Condition object used for comparisons. Results in boolean value.
008 *
009 * @author joel
010 */
011public final class Condition {
012
013    /**
014     * Determines whether the statement is a condition.
015     *
016     * @param value the statement to test
017     * @return whether the statement is a condition
018     */
019    public static boolean isCondition(String value) {
020        return StringUtils.contains(value, "==") || StringUtils.contains(value, "!=")
021                || StringUtils.contains(value, ">") || StringUtils.contains(value, "<")
022                || Value.isBoolean(value);
023    }
024
025    /**
026     * Returns a condition based on it's pure value. Does not change values once
027     * this method is called. Must recall this method to update value.
028     *
029     * @param value value of the condition
030     * @return the condition object representing the condition
031     */
032    public static Condition getConditionFrom(String value) {
033        return new Condition(value);
034    }
035    private final String statement;
036
037    private Condition(String condition) {
038        this.statement = condition;
039    }
040
041    /**
042     * Transfers the condition to a string.
043     *
044     * @return statement (condition)
045     */
046    public String toString() {
047        return statement;
048    }
049
050    /**
051     * Returns if the condition is true.
052     *
053     * @return if true
054     */
055    public boolean isTrue() {
056        Value value1, value2;
057        char test;
058        boolean equals;
059        if (StringUtils.contains(statement, "!=")) {
060            String v1 = statement.substring(0, statement.indexOf("!=")).trim();
061            String v2 = statement.substring(statement.indexOf("!=") + 2).trim();
062            value1 = new Value(v1);
063            value2 = new Value(v2);
064            test = '=';
065            equals = false;
066        } else if (StringUtils.contains(statement, "==")) {
067            String v1 = statement.substring(0, statement.indexOf("==")).trim();
068            String v2 = statement.substring(statement.indexOf("==") + 2).trim();
069            value1 = new Value(v1);
070            value2 = new Value(v2);
071            test = '=';
072            equals = true;
073        } else if (StringUtils.contains(statement, ">=")) {
074            String v1 = statement.substring(0, statement.indexOf(">=")).trim();
075            String v2 = statement.substring(statement.indexOf(">=") + 2).trim();
076            value1 = new Value(v1);
077            value2 = new Value(v2);
078            test = '>';
079            equals = true;
080        } else if (StringUtils.contains(statement, "<=")) {
081            String v1 = statement.substring(0, statement.indexOf("<=")).trim();
082            String v2 = statement.substring(statement.indexOf("<=") + 2).trim();
083            value1 = new Value(v1);
084            value2 = new Value(v2);
085            test = '<';
086            equals = true;
087        } else if (StringUtils.contains(statement, "<") && !StringUtils.contains(statement, "=")) {
088            String v1 = statement.substring(0, statement.indexOf("<")).trim();
089            String v2 = statement.substring(statement.indexOf("<") + 1).trim();
090            value1 = new Value(v1);
091            value2 = new Value(v2);
092            test = '<';
093            equals = false;
094        } else if (StringUtils.contains(statement, ">") && !StringUtils.contains(statement, "=")) {
095            String v1 = statement.substring(0, statement.indexOf(">")).trim();
096            String v2 = statement.substring(statement.indexOf(">") + 1).trim();
097            value1 = new Value(v1);
098            value2 = new Value(v2);
099            test = '>';
100            equals = false;
101        } else if (statement.equalsIgnoreCase("true")
102                || new Value(statement).getValue().toString().equalsIgnoreCase("true")) {
103            return true;
104        } else {
105            return false;
106        }
107        if (test == '=') {
108            // == and equals
109            if ((equals && value1.equals(value2))
110                    // != and not equals
111                    || (!equals && !value1.equals(value2))) {
112                return true;
113            } else {
114                return false;
115            }
116        } else if (test == '>') {
117            // Needs to be a number value. Throws error if not.
118            double double1 = Double.parseDouble(value1.getValue().toString());
119            double double2 = Double.parseDouble(value2.getValue().toString());
120            if (equals) {
121                if (value1.equals(value2) || double1 > double2) {
122                    // >= (== or >)
123                    return true;
124                } else {
125                    // Does not satify >=
126                    return false;
127                }
128            } else {
129                if (double1 > double2) {
130                    // >
131                    return true;
132                } else {
133                    // Does not satisfy >
134                    return false;
135                }
136            }
137        } else if (test == '<') {
138            // Boilerplate
139            double double1 = Double.parseDouble(value1.getValue().toString());
140            double double2 = Double.parseDouble(value2.getValue().toString());
141            if (equals) {
142                if (value1.equals(value2) || double1 < double2) {
143                    return true;
144                } else {
145                    return false;
146                }
147            } else {
148                if (double1 < double2) {
149                    return true;
150                } else {
151                    return false;
152                }
153            }
154        } else {
155            return false;
156        }
157    }
158}
